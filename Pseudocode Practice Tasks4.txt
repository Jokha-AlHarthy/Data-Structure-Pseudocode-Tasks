Task 136: Password Strength Checker

START
	BEGIN
		DISPLAY "Enter your password: "
		READ password
		IF LENGTH OF password LESS THAN 8 TEHN
			DISPLAY "Weak"
		ELSE
			DISPLAY "Strong"
		END IF
	END
STOP

Task 137: First Character Identifier

START
	BEGIN
		DISPLAY "Enter any words"
		READ word
		IF LENGTH OF word EQUALS TO 0 THEN
			DISPLAY "No character"
		ELSE 
			DISPLAY CHARACTER AT TEXT INDEX 0
		END IF
	END
STOP
