Task 182: Extract All Uppercase Letters 
START
	BEGIN
		DISPLAY "Enter a sentence"
		READ sentence
		SET sentence TO ""
		SET result TO ""
		REPEATE FROM I EQUALS TO LENGTH OF sentence
			SET char TO CHARECTER AT sentence INDEX I
			IF IS UPPERCASE char THEN
				SET result TO result + char
			END IF
		END REPEAT
		DISPLAY result
		
	END
STOP

Task 184: Replace All Digits with '#' 
START
	BEGIN
		DISPLAY "Enter a sentence"
		READ sentence
		SET sentence TO ""
		SET result TO ""
		REPEAT FROM I EQUALS TO LENGTH OF sentence
			SET char TO CHARACTER AT sentence INDEX I
			IF IS DIGIT char THEN
				SET result TO result + "#"
			ELSE 
				SET result TO result + char
			END IF
		END REPEAT
		DISPLAY result
	END
STOP

Task 185: Check if String Contains Only Uppercase Letters 
START
	BEGIN
		DISPLAY "Enter a sentence"
		READ sentence
		SET sentence TO ""
		SET result TO ""
		SET onlyUpper TO TRUE
		REPEAT FROM I EQUALS TO LENGTH OF sentence
			SET char TO CHARACTER AT sentence INDEX I
			IF NOT(char GRATER THAN OR EQUAL TO "A" AND char LESS THAN OR EQUAL TO "Z") THEN
				SET onlyUpper TO FLASE
			END IF
		END REPEAT
		IF onlyUpper EQUALS TURE THEN
			DISPLAY "All uppercase letters"
		ELSE
			DISPLAY "Not all uppercase letters"
		END IF
	END
STOP
