Task 182: Extract All Uppercase Letters 
START
	BEGIN
		DISPLAY "Enter a sentence"
		READ sentence
		SET sentence TO ""
		SET result TO ""
		REPEATE FROM I EQUALS TO LENGTH OF sentence
			SET char TO CHARECTER AT sentence INDEX I
			IF IS UPPERCASE char THEN
				SET result TO result + char
			END IF
		END REPEAT
		DISPLAY result
		
	END
STOP

Task 185: Check if String Contains Only Uppercase Letters 
START
	BEGIN
		DISPLAY "Enter a sentence"
		READ sentence
		SET sentence TO ""
		SET onlyUpper TO TRUE
		REPEAT FROM I EQUALS TO LENGTH OF sentence
			SET char TO CHARACTER AT sentence INDEX I
			IF NOT(char GRATER THAN OR EQUAL TO "A" AND char LESS THAN OR EQUAL TO "Z") THEN
				SET onlyUpper TO FLASE
			END IF
		END REPEAT
		IF onlyUpper EQUALS TURE THEN
			DISPLAY "All uppercase letters"
		ELSE
			DISPLAY "Not all uppercase letters"
		END IF
	END
STOP

Task 190: Find Middle Character 
START
	BEGIN
		DISPLAY "Enter a sentence"
		READ sentence
		SET sentence TO ""
		SET len TO 0
		SET len TO LENGTH OF sentence
		IF len MOD 2 EQUALS 1 THEN
			SET mid TO (len / 2 )+1
			DISPLAY CHARACTER AT sentence INDEX mid
		ELSE 
			SET mid1 TO len / 2
			SET mid2 TO mid1+1
			DISPLAY  CHARCTER AT sentence INDEX mid1 + CHARACTER AT sentence INDEX mid2
		END IF
	END
STOP

Task 191: Capitalize First Character of String 
START
	BEGIN
		DISPLAY "Enter a sentence"
		READ sentence
		SET sentence TO ""
		SET result TO ""
		IF LENGTH OF sentence GREATER THAN 0 TEHN
			SET first TO CHARCTER AT sentence INDEX 1
			IF IS LETTER first THEN 
				SET first TO UPPERCASE OF first 
			END IF
			SET rest TO sentence FROM INDEX 2 TO LENGTH OF sentence
			SET result TO first + first
		ELSE 
			SET result TO sentence
		END IF
		DISPLAY result
	END
STOP

Task 193: Count Occurrences of Substring 
START
	BEGIN
		DISPLAY "Enter a sentence"
		READ sentence
		DISPLAY "Enter a substring"
		READ substring
		SET sentence TO""
		SET substring TO ""
		SET position TO 0
		SET position TO FIRST INDEX OF substring IN sentence
		REPEAT WHILE position NOT EQULAS TO0
			SET count TO count+1
			SET position TO FIRST INDEX OF substring IN sentence FROM INDEX position + 1
		END REPEAT
		DISPLAY count 
	END
STOP

Task 194: Check if String Contains Only Digits 
START
	BEGIN
		DISPLAY "Enter a sentence"
		READ sentence
		SET sentence TO ""
		SET onlyDigits TO TRUE
		REPEAT FROM I EQULAS 1 TO LENGTH OF sentence 
			SET char TO CHARACTER AT sentence INDEX I

				SET onlyDigits TO FALSE
			END IF
		END REPEAT
		IF onlyDigits EQULAS TRUE THEN
			DISPLAY "Only digits"
		ELSE
			DISPLAY "Contains non-digits"

	END

Task 195: Remove Duplicate Consecutive Characters 
START
	BEGIN
		DISPLAY "Enter a sentence"
		READ sentence
		SET sentence TO ""
		SET result TO ""
		SET remove TO ""
		REPEAT FROM I EQUALS 1 TO LENGTH OF sentence
			SET char TO CHARACTER AT sentence INDEX I
			IF char NOT EQULAS remove + char
				SET result TO result + char
				SET remove TO char
			END IF
		END REPEAT
		DISPLAY result
	END
STOP
