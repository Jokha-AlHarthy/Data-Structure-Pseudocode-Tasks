Task 26:
START
	BEGIN
		SET QUALITY_CHECK = "PENDING"
		DISPLAY "Enter wight error:"
		READ WEIGHT_ERRORS
		DISPLAY "Enter Dimension errors:”
		READ DIMENSION_ERRORS
		IF WEIGHT_ERRORS EQUALS TO 0 AND DIMENSION_ERRORS EQUALS TO 0 THEN
			 SET QUALITY_CHECK TO "PASSED"
		ELSEIF  QUALITY_CHECK TO EQUALS "PASSED" THEN
			DISPLAY "Route to Packaging"
		ELSE
			DISPLAY "Route to Scrap Yard"
		END IF
	END
STOP
	
Task 27:
START
	BEGIN
		DISPLAY "Enter room temperature: "
		READ TEMPERATURE
		DISPLAY "
		READ SYSTEM_OVERRIDE
		IF TEMPERATURE GREATER THAN 45 AND SYSTEM_OVERRIDE NOT EQUALS TO "FORCE_OFF" THEN
			DISPLAY "Activate Emergency Cooling"
		ELSEIF TEMPERATURE LESS THAN 5 AND SYSTEM_OVERRIDE NOT EQUALS TO "FORCE_OFF" THEN
			DISPLAY "Activate Emergency Heating"
		ELSE
			DISPLAY "Don't Activate Emergency Heating"
		END IF
	END
STOP


Task 28:
START
	BEGIN
		DISPLAY "Enter home buyer's credit score: "
		READ CREDIT_SCORE
		DISPLAY "Enter down payment amount: "
		READ DOWN_PAYMENT
		IF CREDIT_SCORE GREATER THAN OR EQUALS TO 700 AND DOWN_PAYMENT GREATER THAN OR EQUALS TO 20000 THEN
			DISPLAY "Loan Approved: Low Interest"
		ELSEIF CREDIT_SCORE GREATER THAN OR EQUALS TO 700 AND DOWN_PAYMENT LESS THAN 20000 THEN
			DISPLAY "Loan Approved: Standard Interest"
		ELSE
 			DISPLAY "Loan Denied"
		END IF
	
	END
STOP

Task 29:
START
