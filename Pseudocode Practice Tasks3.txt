START
	BEGIN

		REPEAT FROM I = 1 TO 6
			IF  I GREATER THAN 3 THEN

	END

Task 57:

START
	BEGIN
		SET VARIANCE_TOTAL TO 0
		DISPLAY "Enter actual size of manufacturing dimension: "
		SET IDEAL_SIZE TO 50
		REPEAT FROM I = 1 TO 4
			READ ACTUAL_SIZE
			IF ACTUAL_SIZE GREATER THAN IDEAL_SIZE THEN
				SET DEVIATION = ACTUAL_SIZE - IDEAL_SIZE
			ELSE IF IDEAL_SIZE GREATER THAN ACTUAL_SIZE
				SET DEVIATION = IDEAL_SIZE - ACTUAL_SIZE
			END IF
			ADD DEVIATION TO VARIANCE_TOTAL
		END REPEAT
		DISPLAY "The running balance variance is: " + VARIANCE_TOTAL
	END
STOP

	BEGIN
		DISPLAY "Enter the land's unit as COMMERCIAL, RESIDENTIAL, or AGRICULTURAL: "
		READ ZONE
		DISPLAY " Enter the market valuation: "
		READ VALUATION
		IF ZONE EQUALS "COMMERCIAL" THEN
			SET VALUATION = VALUATION * 0.04
		ELSE IF ZONE EQUALS "RESIDENTIAL"
			IF VALUATION GREATER THAN 300000 THEN
				SET VALUATION = VALUATION * 0.02
			ELSE
				SET VALUATION = VALUATION * 0.01
			END IF
		ELSE IF ZONE EQUALS "AGRICULTURAL" THEN
			SET VALUATION = VALUATION * 0.005
		END IF
	END
