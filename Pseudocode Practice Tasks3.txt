Category 12: Advanced Flag Systems & Resource Tracking

Task 51:

START
	BEGIN
		DISPLAY "Enter the soil moisture (number): "
		READ SOIL_MOISTURE
		DISPLAY "Enter the rain prediction as YES or NO: "
		READ RAIN_PREDICTION
		IF SOIL_MOISTURE LESS THAN 30 AND RAIN_PREDICTION EQUALS "NO THEN
			DISPLAY "Activate Zone 1 and Zone 2 Sprinklers"
		ELSE IF SOIL_MOISTURE LESS THAN 30 AND RAIN_PREDICTION EQUALS "YES" THEN
			DISPLAY "Delay Watering - Rain Expected"
		ELSE
			DISPLAY "System Idle"
		END IF
	END
STOP

Task 52:

START
	BEGIN
		SET COOLING_PUMPS TO "NORMAL"
		DISPLAY "Enter the core temperature:"
		READ CORE_TEMPERATURE
		DISPLAY "Enter humidity level: "
		READ HUMIDITY_LEVEL
		IF CORE_TEMPERATURE GREATER THAN 85 OR HUMIDITY_LEVEL GREATER THAN 70 THEN
			SET COOLING_PUMPS TO "MAXIMUM"
			IF COOLING_PUMPS EQUALS "MAXIMUM" THEN
				DISPLAY "Overclocking Disabled - Cooling Engaged"
			END IF
		END IF
	END
STOP

Category 13: Mathematical Compounding & Sequence Filters

Task 55:

START
	BEGIN
		SET TOTAL_VALUE TO 10
		REPEAT FROM I = 1 TO 6
			IF  I GREATER THAN 3 THEN
				SET TOTAL_VALUE = TOTAL_VALUE * 3
			ELSE IF I LESS THAN OR EQUASL TO 3
				SET TOTAL_VALUE = TOTAL_VALUE * 2
			END IF
		END REPEAT
		DISPLAY "The total value is: " + TOTAL_VALUE
	END
STOP

Task 56:

START
	BEGIN
		SET ACCUMULATED_SALES TO 0
		DISPLAY "Enter the sales amount: "
		REPEAT FROM I = 1 TO 5
		READ SALE_AMOUNT
			SET ACCUMULATED_SALES = ACCUMULATED_SALES + SALE_AMOUNT
		END REPEAT
		IF ACCUMULATED_SALES GREATER THAN 10000 THEN
			SET BONUS = ACCUMULATED_SALES * 0.10
		ELSE
			SET BONUS = ACCUMULATED_SALES * 0.05
		END IF
		DISPLAY BONUS
	
	END
STOP

Task 57:

START
	BEGIN
		SET VARIANCE_TOTAL TO 0
		DISPLAY "Enter actual size of manufacturing dimension: "
		SET IDEAL_SIZE TO 50
		REPEAT FROM I = 1 TO 4
			READ ACTUAL_SIZE
			IF ACTUAL_SIZE GREATER THAN IDEAL_SIZE THEN
				SET DEVIATION = ACTUAL_SIZE - IDEAL_SIZE
			ELSE IF IDEAL_SIZE GREATER THAN ACTUAL_SIZE
				SET DEVIATION = IDEAL_SIZE - ACTUAL_SIZE
			END IF
			ADD DEVIATION TO VARIANCE_TOTAL
		END REPEAT
		DISPLAY "The running balance variance is: " + VARIANCE_TOTAL
	END
STOP

Task 58:

START
	BEGIN
		DISPLAY "Enter starting monetary asset base: "
		READ INVESTMENT
		DISPLAY "Enter variable performance as BULL or BEAR: "
		READ MARKET_TREND
		REPEAT FROM I=1 TO 3
			IF MARKET_TREND EQUALS "BULL" THEN
				SET INVESTMENT = INVESTMENT + (INVESTMENT * 0.12)
			ELSE IF MARKET_TREND EQUALS "BEAR" THEN
				SET INVESTMENT = INVESTMENT - (INVESTMENT * 0.04)
			END IF
		END REPEAT
		DISPLAY "The final assets value is: " + INVESTMENT
	
	END
STOP
