TASK 1:

		DISPLAY "Enter your favorite color"
		READ COLOR

	END
STOP

Category 2: Conditions & Comparisons

Task 5:
START
	BEGIN
		DIAPLY "Enter your Age: "
		IF Age GREATER THAN OR EQUAL 18 THEN
			DISPLAY "Allowed"
		ELSE
			DISPLAY "Not Allowed"
		END IF
	END
STOP

Task 6:
START
	BEGIN
		DISPLAY "Enter a number: "
		READ VAL
		IF VAL GREATER THAN 0 THEN
			DISPLAY "Positive"
		ELSE IF VAL LESS THAN 0 THEN
			DISPLAY "Negative"
		ELSE
			DISPLAY "Zero"
		END IF
	END
STOP

Task 8:
START
	BEGIN
		DIAPLY "Enter the cost price: "
		READ COST_PRICE
		DISPLAY "Enter the second selling price: "
		READ SELLING_PRICE
		IF SELLING_PRICE GREATER THAN COST_PRICE THEN
			DISPLAY "Profit"
		ELSE
			DISPLAY "Loss"
		END IF
	END
STOP


Task 9:

Category 3: Logical Connectors & Nested Conditions

Task 10:
START
	BEGIN
		DIAPLY "Enter the student that has attendance: "
		READ HAS_ATTENDANCE
		DISPLAY "Enter the student that has assignment: "
		READ HAS_ASSIGNMENT
		IF  HAS_ATTENDANCE EQUALS "YES" AND HAS_ASSIGNMENT EQUALS "YES" THEN
			DISPLAY "Exam Allowed"
		ELSE
			DISPLAY "Exam Barred"
		END IF
	END
STOP

Task 11:
START
	BEGIN
		DIAPLY "Enter the student's GPA: "
		READ GPA
		DISPLAY "Enter the student's family income: "
		READ Income
		IF GPA GREATER THAN OR EQUAL TO 3.5 OR Income LESS THAN 20000 THEN
			DISPLAY "Scholarship Approved""
		ELSE
			DISPLAY "Scholarship Denied"
		END IF
	END
STOP

Task 12:
START
	BEGIN
		DIAPLY "Enter the number of the item: "
		READ NoItems
		IF NoItems GREATER THAN OR EQUAL TO 0 AND NoItems LESS THAN OR EQUAL TO 100 THEN
			DISPLAY "Valid Score"
		ELSE
			DISPLAY "Invalid Score"
		END IF
	END
STOP

Task 13:

	BEGIN
		DIAPLY "Enter your age: "

STOP
