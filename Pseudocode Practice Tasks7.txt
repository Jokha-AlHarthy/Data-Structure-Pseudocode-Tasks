Task 201: Event Check-In System
START
	BEGIN
		SET TICKET_TYPES TO MAP ["VIP001" maps to "VIP", "GEN002" maps to "General"]
		CREATE SET CHECKED_IN 
		READ GUEST_NAME
		READ TICKET_CODE
		SET TICKET_CODE TO UPPERCASE TICKET_CODE
		IF SET CONTAINS CHECKED_IN IN TICKET_CODE THEN
			DISPLAY "The ticket already used & the entry refused"
		ELSE IF MAP CONTAINS KEY TICKET_TYPES THEN
			ADD TICKET_CODE TO SET CHECKED_IN
			SET TIER TO GET FROM MAP TICKET TYPES TICKET_CODE
			DISPLAY "Welcome, " + GUEST_NAME + " - " + TIER + "access granted"
		ELSE
			DISPLAY "Sorry.. ticket is not recognized"
		END IF
	END 
STOP

Task 202: Student Subject Enrolment

START
	BEGIN
		SET CAPACITY TO MAP ["Math" TO 30, "Science" TO 25]
		CREATE SET ENROLLED 
		SET LOG TO LIST []
		READ SUBJECT
		REPEAT WHILE NOT SUBJECT EQULAS "DONE"
			SET SUBJECT TO TIRM SUBJECT
			IF SET CONTAINS ENROLLED SUBJECT THEN
				DISPLAY "The subject is enrolled" + SUBJECT
			ELSE IF NOT MAP CONTAINS KEY CAPACITY SUBJECT THEN
				DISPLAY "The subject is not enrolled"
			ELSE IF GET FROM MAP CAPACITY SUBJECT EQUALS 0 THEN
				DISPLAY "There is no seat remain for" + SUBJECT
			ELSE
				ADD SUBJECT TO SET ENROLLED 
				ADD SUBJECT TO LIST LOG
				SET MAP CAPACITY SUBJECT TO (GET FROM MAP  CAPACITY SUBJECT ) - 1
				DISPLAY "Enrolled in" + SUBJECT
			END IF
			READ SUBJECT
		END REPEAT
		DISPLAY "The total subjects enrolled:+ SIZE OF LOG
	END 
STOP

Task 203: Username Registration Checker

START
    BEGIN

        SET USER_DB TO MAP ["ali99" TO "Ali Hassan", "sara_m" TO "Sara Ali"]
        DISPLAY "Enter username:"
        READ USERNAME
        DISPLAY "Enter full name"
        READ FULLNAME
        SET USERNAME TO LOWERCASE(TRIM USERNAME)
        IF LENGTH OF USERNAME LESS THAN 4 THEN 
            DISPLAY "Username is too short"
        ELSE IF LENGTH OF USERNAME GREATER THAN 15 THEN
            DISPLAY "Username is too long"
        ELSE IF MAP CONTAINS KEY USER_DB USERNAME THEN
            DISPLAY "Username already exists."
        ELSE
            SET MAP USER_DB USERNAME TO FULLNAME

            DISPLAY "Account created successfully."
            DISPLAY "Total users = " + MAP SIZE OF USER_DB

        END IF
    END
STOP

            END IF

        DISPLAY "Unique borrowers is " + SET SIZE OF BORROWERS

Task 206: Shopping Cart & Discount Engine
START
    BEGIN

        SET PRICES TO MAP ["Laptop" TO 800, "Phone" TO 400, "Headphones" TO 120]
        CREATE SET SEEN
        SET CART TO LIST []
        SET TOTAL TO 0
        DISPLAY "Enter item "
        READ ITEM
        REPEAT WHILE ITEM NOT EQUALS "CHECKOUT"
            SET ITEM TO (TRIM ITEM)
            IF NOT MAP CONTAINS KEY PRICES ITEM THEN
                DISPLAY "Item not found"
            ELSE IF SET CONTAINS SEEN ITEM THEN
                DISPLAY "Item already added"
            ELSE
                ADD ITEM TO SET SEEN
                ADD ITEM TO LIST CART
                SET TOTAL TO TOTAL + GET FROM MAP PRICES ITEM
                DISPLAY "Item added"
            END IF
            DISPLAY "Enter item "
            READ ITEM
        END REPEAT
        IF SIZE OF CART GREATER THAN OR EQUAL TO 3 THEN
            SET TOTAL TO TOTAL * 0.90
            DISPLAY "10% discount applied"
        END IF
        DISPLAY "The total is" + TOTAL
    END
STOP

Task 207: Employee Payroll Calculator 

START
    BEGIN

        SET RATES TO MAP ["Engineering" TO 50, "Marketing" TO 40, "Support" TO 30]
        CREATE SET PAID_IDS
        SET GRAND_TOTAL TO 0
        REPEAT FROM I EQULAS 1 TO 4 
            DISPLAY "Enter employee ID"
            READ EMP_ID
            DISPLAY "Enter employee name"
            READ NAME
            DISPLAY "Enter department"
            READ DEPARTMENT
            DISPLAY "Enter hours worked"
            READ HOURS
            SET NAME TO (TRIM NAME)
            IF SET CONTAINS PAID_IDS EMP_ID THEN
                DISPLAY "Employee already paid"
            ELSE IF NOT MAP CONTAINS KEY RATES DEPARTMENT THEN
                DISPLAY "Department not found"
            ELSE
                SET GROSS TO HOURS * GET FROM MAP RATES DEPARTMENT
                SET NET TO GROSS - (GROSS * 0.15)
                ADD EMP_ID TO SET PAID_IDS
                SET GRAND_TOTAL TO GRAND_TOTAL + NET
                DISPLAY "Net Pay = " + NET
            END IF
        END REPEAT
        DISPLAY "The total Payroll = " + GRAND_TOTAL
    END
STOP

        DISPLAY "The total of tax is" + TOTAL_TAX
        DISPLAY "Categories used size is" + SET SIZE OF CATS_USED

Task 209: Bank Account Transaction Processor

START
    BEGIN

        SET ACCOUNTS TO MAP["ACC001" TO 1500,"ACC002" TO 300]
        CREATE SET FLAGGED
        SET LOG TO LIST[]
        REPEAT FROM I EQULAS 1 TO 4
            DISPLAY "Enter account ID"
            READ ACCOUNT
            DISPLAY "Enter operation"
            READ OPERATION
            DISPLAY "Enter amount"
            READ AMOUNT
            SET OPERATION TO UPPERCASE OPERATION
            IF NOT MAP CONTAINS KEY ACCOUNTS ACCOUNT THEN
                DISPLAY "Account not found"
            ELSE IF AMOUNT LESS THAN OR EQUAL TO 0 THEN
                DISPLAY "Invalid amount"
            ELSE IF SET CONTAINS FLAGGED ACCOUNT THEN
                DISPLAY "Account is blocked"
            ELSE
                IF OPERATION EQUALS "DEPOSIT" THEN
                    SET MAP ACCOUNTS ACCOUNT TO GET FROM MAP ACCOUNTS ACCOUNT + AMOUNT
                ELSE IF OPERATION EQUALS "WITHDRAW" THEN
                    IF AMOUNT GREATER THAN GET FROM MAP ACCOUNTS ACCOUNT THEN
                        DISPLAY "Insufficient balance"
                    ELSE
                        SET MAP ACCOUNTS ACCOUNT TO GET FROM MAP ACCOUNTS ACCOUNT - AMOUNT
                        IF AMOUNT GREATER THAN 2000 THEN
                            ADD ACCOUNT TO SET FLAGGED
                        END IF
                    END IF
                END IF
                ADD ACCOUNT TO LIST LOG
                DISPLAY "The balance is " + GET FROM MAP ACCOUNTS ACCOUNT
            END IF
        END REPEAT

    END
STOP

Task 210: Investment Portfolio Updater

START
    BEGIN

        SET PORTFOLIO TO MAP["Gold" TO 5000, "Stocks" TO 8000]
        CREATE SET GAINERS
        SET REPORT TO LIST[]
        SET TOTAL TO 0
        SET KEYS TO GET ALL KEYS PORTFOLIO
        REPEAT FROM I EQUALS 1 TO SIZE OF KEYS
            DISPLAY "Enter percentage change"
            READ CHANGE
            SET VALUE TO GET FROM MAP PORTFOLIO KEYS[I]
            SET NEW_VALUE TO VALUE + (VALUE * CHANGE)
            SET MAP PORTFOLIO KEYS[I] TO NEW_VALUE
            ADD KEYS[I] TO LIST REPORT
            SET TOTAL TO TOTAL + NEW_VALUE
            IF CHANGE GREATER THAN 0 THEN
                ADD KEYS[I] TO SET GAINERS
            END IF
            DISPLAY KEYS[I] + " Updated"
        END REPEAT
        DISPLAY "The total value is" + TOTAL
        DISPLAY "Gainers size is " + SET SIZE OF GAINERS

    END
STOP

Task 211: Login System with Lockout
START
    BEGIN
        SET USER_DB TO MAP["admin" TO "Admin@99","sara" TO "Sara#25"]
        SET FAILS TO LIST[]
        CREATE SET LOCKED
        SET AUDIT TO LIST[]
        REPEAT FROM I EQAULS 1 TO 3
            DISPLAY "Enter username"
            READ USERNAME
            DISPLAY "Enter password"
            READ PASSWORD
            SET USERNAME TO LOWERCASE(TRIM USERNAME)
            IF SET CONTAINS LOCKED USERNAME THEN
                DISPLAY "Account is locked"
            ELSE IF NOT MAP CONTAINS KEY USER_DB USERNAME THEN
                DISPLAY "Username not found"
            ELSE IF GET FROM MAP USER_DB USERNAME EQUALS PASSWORD THEN
                DISPLAY "Login successful"
                ADD USERNAME TO LIST AUDIT
            ELSE
                IF NOT MAP CONTAINS KEY FAILS USERNAME THEN
                    SET MAP FAILS USERNAME TO 1
                ELSE
                    SET MAP FAILS USERNAME TO GET FROM MAP FAILS USERNAME + 1
                END IF
                IF GET FROM MAP FAILS USERNAME GREATER THAN OR EQUAL TO 3 THEN
                    ADD USERNAME TO SET LOCKED
                    DISPLAY "Account locked"
                ELSE
                    DISPLAY "Wrong password"
                END IF
                ADD USERNAME TO LIST AUDIT
            END IF
        END REPEAT
        DISPLAY "The audit records size is" + SIZE OF AUDIT
    END
STOP
