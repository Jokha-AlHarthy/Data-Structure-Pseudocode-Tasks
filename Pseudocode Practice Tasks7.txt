Task 201: Event Check-In System
START
	BEGIN
		SET TICKET_TYPES TO MAP ["VIP001" maps to "VIP", "GEN002" maps to "General"]
		CREATE SET CHECKED_IN 
		READ GUEST_NAME
		READ TICKET_CODE
		SET TICKET_CODE TO UPPERCASE TICKET_CODE
		IF SET CONTAINS CHECKED_IN IN TICKET_CODE THEN
			DISPLAY "The ticket already used & the entry refused"
		ELSE IF MAP CONTAINS KEY TICKET_TYPES THEN
			ADD TICKET_CODE TO SET CHECKED_IN
			SET TIER TO GET FROM MAP TICKET TYPES TICKET_CODE
			DISPLAY "Welcome, " + GUEST_NAME + " - " + TIER + "access granted"
		ELSE
			DISPLAY "Sorry.. ticket is not recognized"
		END IF
	END 
STOP

Task 202: Student Subject Enrolment

START
	BEGIN
		SET CAPACITY TO MAP ["Math" TO 30, "Science" TO 25]
		CREATE SET ENROLLED 
		SET LOG TO LIST []
		READ SUBJECT
		REPEAT WHILE NOT SUBJECT EQULAS "DONE"
			SET SUBJECT TO TIRM SUBJECT
			IF SET CONTAINS ENROLLED SUBJECT THEN
				DISPLAY "The subject is enrolled" + SUBJECT
			ELSE IF NOT MAP CONTAINS KEY CAPACITY SUBJECT THEN
				DISPLAY "The subject is not enrolled"
			ELSE IF GET FROM MAP CAPACITY SUBJECT EQUALS 0 THEN
				DISPLAY "There is no seat remain for" + SUBJECT
			ELSE
				ADD SUBJECT TO SET ENROLLED 
				ADD SUBJECT TO LIST LOG
				SET MAP CAPACITY SUBJECT TO (GET FROM MAP  CAPACITY SUBJECT ) - 1
				DISPLAY "Enrolled in" + SUBJECT
			END IF
			READ SUBJECT
		END REPEAT
		DISPLAY "The total subjects enrolled:+ SIZE OF LOG
	END 
STOP

Task 203: Username Registration Checker

START
    BEGIN

        SET USER_DB TO MAP ["ali99" TO "Ali Hassan", "sara_m" TO "Sara Ali"]
        DISPLAY "Enter username:"
        READ USERNAME
        DISPLAY "Enter full name"
        READ FULLNAME
        SET USERNAME TO LOWERCASE(TRIM USERNAME)
        IF LENGTH OF USERNAME LESS THAN 4 THEN 
            DISPLAY "Username is too short"
        ELSE IF LENGTH OF USERNAME GREATER THAN 15 THEN
            DISPLAY "Username is too long"
        ELSE IF MAP CONTAINS KEY USER_DB USERNAME THEN
            DISPLAY "Username already exists."
        ELSE
            SET MAP USER_DB USERNAME TO FULLNAME

            DISPLAY "Account created successfully."
            DISPLAY "Total users = " + MAP SIZE OF USER_DB

        END IF
    END
STOP

Task 204: Library Book Loan System 

START
    BEGIN

        SET STOCK TO MAP ["Dune" TO 3,"Hamlet" TO 0]
        CREATE SET BORROWERS
        SET LOANS TO LIST []

        REPEAT FROM I EQUALS 1 TO 4 
            DISPLAY "Enter member name"
            READ MEMBER
            DISPLAY "Enter book title"
            READ BOOK
            SET MEMBER TO (TRIM MEMBER)
            SET BOOK TO (TRIM BOOK)
            IF NOT MAP CONTAINS KEY STOCK BOOK THEN
                DISPLAY "Book not found"+BOOK
            ELSE IF GET FROM MAP STOCK BOOK EQUALS 0 THEN
                DISPLAY "No copies available"
            ELSE
                SET MAP STOCK BOOK TO GET FROM MAP STOCK BOOK - 1
                ADD MEMBER TO SET BORROWERS
                ADD BOOK TO LIST LOANS
                DISPLAY "Loan approved."
            END IF
        END REPEAT
        DISPLAY "Total loans is" + SIZE OF LOANS
        DISPLAY "Unique borrowers is " + SET SIZE OF BORROWERS
    END
STOP

Task 205: Hotel Room Booking Validator

START
    BEGIN
        SET ROOMS TO MAP["101" TO "AVAILABLE","102" TO "BOOKED","103" TO "AVAILABLE"]
        CREATE SET VIP_ROOMS
        ADD "101" TO SET VIP_ROOMS
        DISPLAY "Enter guest name"
        READ GUEST
        DISPLAY "Enter room number"
        READ ROOM
        DISPLAY "Enter membership"
        READ MEMBERSHIP
        SET GUEST TO (TRIM GUEST)
        IF NOT MAP CONTAINS KEY ROOMS ROOM THEN
            DISPLAY "Room does not exist"
        ELSE IF GET FROM MAP ROOMS ROOM EQUALS "BOOKED" THEN
            DISPLAY "Room is already booked"
        ELSE IF SET CONTAINS VIP_ROOMS ROOM AND MEMBERSHIP NOT EQUALS "VIP" THEN
            DISPLAY "VIP membership required"
        ELSE
            SET MAP ROOMS ROOM TO "BOOKED"
            DISPLAY "Room booked successfully"
        END IF

    END
STOP

Task 206: Shopping Cart & Discount Engine
START
    BEGIN

        SET PRICES TO MAP ["Laptop" TO 800, "Phone" TO 400, "Headphones" TO 120]
        CREATE SET SEEN
        SET CART TO LIST []
        SET TOTAL TO 0
        DISPLAY "Enter item "
        READ ITEM
        REPEAT WHILE ITEM NOT EQUALS "CHECKOUT"
            SET ITEM TO (TRIM ITEM)
            IF NOT MAP CONTAINS KEY PRICES ITEM THEN
                DISPLAY "Item not found"
            ELSE IF SET CONTAINS SEEN ITEM THEN
                DISPLAY "Item already added"
            ELSE
                ADD ITEM TO SET SEEN
                ADD ITEM TO LIST CART
                SET TOTAL TO TOTAL + GET FROM MAP PRICES ITEM
                DISPLAY "Item added"
            END IF
            DISPLAY "Enter item "
            READ ITEM
        END REPEAT
        IF SIZE OF CART GREATER THAN OR EQUAL TO 3 THEN
            SET TOTAL TO TOTAL * 0.90
            DISPLAY "10% discount applied"
        END IF
        DISPLAY "The total is" + TOTAL
    END
STOP
